def cssAssembleLoop(arr)      
  arr.each do |cs|

    #  スタイルがない（｛｝のみ）なら標示しない        
    next unless KString.isExistCssString cs.str
    
    # 同じ名前のスタイルは書き込まない（重複書き込み禁止）
    next if isSameCSSName(cs.name)

    cssAssembleCore cs
  end      
end

def usageCSS

      name + {id,cls} + {css}

      p {color:red;}

      addCss a
