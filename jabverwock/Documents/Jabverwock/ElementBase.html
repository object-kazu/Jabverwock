  # units.lastのデータを使用する必要がある
  content = self.docHashLast
  
  if content == ''
    content = self.unitsHashLast
    self.removeLastUnitsHash #unitsデータを使用したので削除しておく
  end
  
  isVarMain name, content
  # v = JsVar.new

end
