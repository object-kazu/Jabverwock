def mainLoop
  while @elmArray.count > 0
    target = @elmArray.first
    
    if KSUtil.isOpenPara(target) || isStyle(target)
      shiftPlusTab
    elsif KSUtil.isClosePara(target)|| isCloseStyle(target)
      minusTabShift
    else
      shiftMain
    end   
  end    
end
