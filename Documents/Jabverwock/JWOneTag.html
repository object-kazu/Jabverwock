def makeResult
  @templeteString = @tagManager.tempOpenString

  if @content.is_a? NilClass
    @content = ""
  end
  
  unless @content == ""
    @templeteString << @content
  end
        
  @templeteString = KString.removeLastRET(@templeteString)
end
