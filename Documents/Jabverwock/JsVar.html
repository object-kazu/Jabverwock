def collection(name, index = -1, *exp)

  if index == -1
    c ="#{name.to_s}"
    @units.update seqHash c
    return self
  end
  
  if exp.count == 0
    c ="#{name.to_s}[#{index}]"
    @units.update seqHash c
    return self
  end
  
  c ="#{name.to_s}[#{index}].#{exp.first}"
  @units.update seqHash c
  self
end
