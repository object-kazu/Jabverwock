def arraying (str)
  # a{b,c{d},e},f
  # "{" => "_c_", "}" => "_e_", "," => "_m_"
  # => ["a", "_c_", "b", "_m_", "c","_c_","d", "_e_","_m_", "e","_e_", "_m_", "f"]
  ans = str.gsub(/{/," #{$CHILD} ").gsub(/,/," #{$MEMBER} ").gsub(/}/," #{$ENDMARK} ")
  ans.split()
end

def coden(l, c, r)
  case c
  when $CHILD
    addChild l,r
    recordParent l
  when $MEMBER
    addMember l,r
  when $ENDMARK
    parIndex += 1
    endMarkCoden r
  end
end
