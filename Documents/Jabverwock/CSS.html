def nameTreatment(name)
  @name = ""
  if name.is_a? String
    @name = name
    return
  end

  if name.is_a? Symbol
    symbolName name        
    return
  end

  @name = name.to_s
  
end

def symbolName(name)
  if KSUtil.isDoubleUnderBarSymbol name
    names = name.divid
    
    if KSUtil.isID names[0]
      idSymbol names[1]
      return
    end
    
    if KSUtil.isCls names[0]
      classSymbol names[1]
      return
    end
    
    # remove "_ _"        
    @name = removeDoubleUnderBar name
    return
  end
  @name = name.to_s
end
