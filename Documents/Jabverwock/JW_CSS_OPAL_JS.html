def addOpalScriptTag

  endTag = "</script>"
  @scriptTag << startTag << endTag << $RET

def readOpalFile(path)
  @opalPath = path
  op = OpalFileReader.new
  op.opalPath = path
  op.readOpal
  @opalFileName = op.opalFileName
end
